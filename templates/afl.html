function _vfApplyFilters(data) { // Existing filters for line and edge data = data.filter(b => b.line >= minLine && b.line <= maxLine); data = data.filter(b => b.edge >= minEdge); // New filter to only show VALUE bets data = data.filter(b => b.rec === 'value' || b.reco === 'VALUE' || b.is_value === true); // If no value bets are found, show appropriate message if (data.length === 0) { displayEmptyState(); } return data; }
